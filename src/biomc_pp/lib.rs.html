8/// A struct that wraps the `PostProcess` type for Python bindings.
9///
10/// The `PythonPostProcess` struct is designed to provide a Python interface for the
11/// `PostProcess` struct, enabling the interaction with Rust's `PostProcess` type in Python.
12///
13/// # Fields
14///
15/// - `inner`: A reference to the inner `PostProcess` object, which holds the actual
16///   post-processing logic that can be used in Python.
17///
18/// # Example
19///
20/// ```python
21/// # Example usage in Python
22/// post_process = PostProcess(name,root)
23/// ```
24#[derive(Debug)]
25#[pyclass(name = "PostProcess")]
26struct PythonPostProcess {
27    inner: PostProcess,
28}
